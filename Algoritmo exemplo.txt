# Um exemplo de algoritmo na nova linguagem para gerar um arquivo models.py:

Model-Begin

    Entity-Begin Game
        name: string()
        publisher: string()
    Entity-End

    Entity-Begin Release
        game: Game
        platform: Platform
        releasedate: date
        version: int
    Entity-End

    Entity-Begin Platform
        name: string(max_length=255, unique=True) #attr de params em des.
    Entity-End

Model-End
