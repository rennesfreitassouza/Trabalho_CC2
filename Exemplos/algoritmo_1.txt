# Algoritmo para a criação de uma biblioteca de jogos.

Config-Begin
	SITE: mysite
	APP: gamelib
Config-End

Model-Begin	
	Entity-Begin Games
		name: string(max_length=255, unique)
		publisher: string(max_length=255, unique)
	Entity-End

	Entity-Begin Platforms
		name: string(max_length=255, unique)
	Entity-End

	Entity-Begin Releases
		game: Games
		platform: Platforms
		releasedate: date
		version: int
	Entity-End
Model-End
